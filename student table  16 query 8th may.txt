CREATE FUNCTION add_new_column()
RETURNS TABLE
AS
RETURN
SELECT studentname,class,studentno,game,gradeone,supw,gradetwo,'newvalue' as marks FROM GamesAndSUPWDetails;

SELECT *
INTO new_table
FROM add_new_column();

select *from new_table;
select *from GamesAndSUPWDetails;
